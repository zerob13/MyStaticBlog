struct node{

    friend bool operator< (node n1, node n2)
    {
        return n1.a > n2.a;
    }
};
priority_queuezerob;
int main (int argc, char * const argv[]) {
    int i,j,k,l;
    node kk;
    while(scanf(“%d”,&n)!=EOF){
        while(!zerob.empty()){
            zerob.pop();
        }
        for(i=0;i<n;i++)
        {
            scanf(“%d”,&lef);
            kk.a=lef;
            zerob.push(kk);



    }

    long long int sum=0;
    while(zerob.size()>1)
    {
        node a=zerob.top();
        zerob.pop();
        node b=zerob.top();
        zerob.pop();
        sum+=a.a+b.a;
        kk.a=a.a+b.a;
        zerob.push(kk);
    }
