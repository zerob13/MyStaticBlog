java练习之N皇后深搜解

之前一直脑残，代码也不会写了，最近决定重新开始发奋，记得大一的时候第一个像样的c程序就是八皇后问题。于是就有了今天的N皇后java版本。 不是什么高效的版本，就是最简单最好理解的深搜，无优化，大牛无视。 




package com.zerob13.eightqueen;



/**
 * @author yanglingfeng
 *
 */
public class EightQueen {
    private boolean[][] broad;
    private boolean[] hashc;
    int count;
    private boolean[] hashlx;
    private boolean[] hashrx;
    private int N;
    public EightQueen(int n){
        this.N=n;
        count=0;
        this.broad=new boolean[n][];
        for(int i=0;i<n;i++)
            this.broad[i]=new boolean[n];
        this.hashc=new boolean[n];
        this.hashlx=new boolean[n+n];
        this.hashrx=new boolean[n+n];



}
public EightQueen(){
    int n=8;
    this.N=n;   
    count=0;
    this.broad=new boolean[n][];
    for(int i=0;i<n;i++)
        this.broad[i]=new boolean[n];
    this.hashc=new boolean[n];
    this.hashlx=new boolean[n+n];
    this.hashrx=new boolean[n+n];

}
public int shownum()
{
    return count;
}
private void show(){
    int i,j;
    count++;
    for(i=0;i<N;i++)
    {
        for(j=0;j<N;j++)
        {
            if(broad[i][j])
            System.out.print("Q");
            else
                System.out.print(".");
        }
        System.out.println();
    }
    for(i=0;i<N;i++)
    System.out.print("-");
    System.out.println();
}
public void dfs(int now){
    int i;
    if (now>=N){
        this.show();
        return ;
    }
    for(i=0;i<N;i++)
    {
        if(broad[now][i]==false&&hashc[i]==false&&hashlx[i+now]==false&&hashrx[i-now+N]==false){
            hashc[i]=true;
            hashlx[i+now]=true;
            hashrx[i-now+N]=true;
            broad[now][i]=true;
            dfs(now+1);
            hashc[i]=false;
            hashlx[i+now]=false;
            hashrx[i-now+N]=false;
            broad[now][i]=false;
        }
    }
    return ;

}
