import java.util.;



public class Matrix {



/**
 * @param args
 */
private int[][] mmat;
private int n;
public Matrix(int n){
    this.n=n;
    this.mmat=new int[n][];

    for(int i=0;i<n;i++){
        this.mmat[i]=new int[n];
        for(int j=0;j<n;j++)
        {
            mmat[i][j]=0;
        }
    }
}
public Matrix(int n,int[][] mmat){
    this.n=n;
    this.mmat=new int[n][];

    for(int i=0;i<n;i++){
        this.mmat[i]=new int[n];
        for(int j=0;j<n;j++)
        {
            this.mmat[i][j]=mmat[i][j];
        }
    }
}
public void ChangeValue(){
    Scanner k=new Scanner(System.in);
    for(int i=0;i<this.n;i++)
    {
        for(int j=0;j<this.n;j++)
        {
            this.mmat[i][j]=k.nextInt();

        }

    }
}
public Matrix mult(Matrix b){
    int i,j,k;
    Matrix temp=new Matrix(this.n);
    int N;
    N=this.n;
    if(b.n!=this.n){
        return null;
    }
    for(i=0;i<N;i++)
    {
        for(j=0;j<N;j++)
        {
            for(k=0;k<N;k++)
            {
                temp.mmat[i][j]+=this.mmat[i][k]*b.mmat[k][j];
                //temp.mmat[i][j]%=2;
            }
        }
    }
    return temp;
}

public void printm(){
    for(int i=0;i<this.n;i++)
    {
        for(int j=0;j<this.n;j++)
        {
            System.out.print(this.mmat[i][j]+" ");
        }
        System.out.println();
    }
}




}
