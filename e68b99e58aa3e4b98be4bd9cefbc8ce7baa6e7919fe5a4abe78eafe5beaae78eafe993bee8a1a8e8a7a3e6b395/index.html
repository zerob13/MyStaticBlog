using namespace std;
struct node{
    int data;
    int index;
    node next,*pre;
}*root,*tail;
void del(node a)
{
    node *p;
    p=a->pre;
    p->next=a->next;
    a->next->pre=p;
    cout<index<<endl;
    free(a);
    return ;
}



void add(int aas,int ind)
{
    node *aa=new node;
    aa->data=aas;
    aa->index=ind;
    aa->pre=tail;
    tail->next=aa;
    aa->next=root;
    tail=aa;
    root->pre=tail;
}
int main()
{
    int i;
    int a;
    int n,m;
    m=4;n=6;
    cin>>m>>n;

    cin>>a;

        add(a,i+1);
    }else{
        root=new node;
        tail=root;
        root->data=a;
        root->index=i+1;
        root->next=tail;
        root->pre=tail;
        tail->next=root;
        tail->pre=root;
    }
}
node *tt;
tt=root;
while(n){
